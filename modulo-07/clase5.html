El store — State y Mutations

// store/index.js
import Vue from 'vue'
import Vuex from 'vuex'
import axios from 'axios'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    pokemones: [],
    detalle: null,
    favoritos: [],
    loading: false,
    error: null
  },
  mutations: {
    SET_POKEMONES(state, lista) { state.pokemones = lista },
    SET_DETALLE(state, poke) { state.detalle = poke },
    ADD_FAVORITO(state, poke) { state.favoritos.push(poke) },
    REMOVE_FAVORITO(state, id) {
      state.favoritos = state.favoritos.filter(f => f.id !== id)
    },
    UPDATE_FAVORITO(state, poke) {
      const i = state.favoritos.findIndex(f => f.id === poke.id)
      if (i !== -1) state.favoritos.splice(i, 1, poke)
    },
    SET_LOADING(state, v) { state.loading = v },
    SET_ERROR(state, e) { state.error = e }
  }
})

Actions — Axios contra la PokéAPI

actions: {
  fetchPokemones({ commit }) {
    commit('SET_LOADING', true)
    commit('SET_ERROR', null)
    axios.get('https://pokeapi.co/api/v2/pokemon?limit=20')
      .then(res => {
        const lista = res.data.results.map((p, i) => ({
          id: i + 1,
          name: p.name,
          image: `https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${i + 1}.png`
        }))
        commit('SET_POKEMONES', lista)
      })
      .catch(err => commit('SET_ERROR', err.message))
      .finally(() => commit('SET_LOADING', false))
  },
  fetchDetalle({ commit }, id) {
    commit('SET_LOADING', true)
    axios.get(`https://pokeapi.co/api/v2/pokemon/${id}`)
      .then(res => {
        commit('SET_DETALLE', {
          id: res.data.id, name: res.data.name,
          image: res.data.sprites.front_default,
          types: res.data.types.map(t => t.type.name),
          height: res.data.height, weight: res.data.weight
        })
      })
      .catch(err => commit('SET_ERROR', err.message))
      .finally(() => commit('SET_LOADING', false))
  }
}

Getters — Datos derivados

getters: {
  totalFavoritos: state => state.favoritos.length,

  esFavorito: state => id => {
    return state.favoritos.some(f => f.id === id)
  },

  pokemonesFiltrados: state => busqueda => {
    return state.pokemones.filter(p =>
      p.name.includes(busqueda.toLowerCase())
    )
  }
}

esFavorito retorna una función: se usa como esFavorito(pokemon.id) para pintar el botón de estrella.

mapHelpers en los componentes

Usamos mapState, mapGetters, mapActions y mapMutations desde el inicio para código limpio.

import { mapState, mapGetters, mapActions, mapMutations } from 'vuex'

export default {
  computed: {
    ...mapState(['pokemones', 'detalle', 'favoritos', 'loading', 'error']),
    ...mapGetters(['totalFavoritos', 'esFavorito', 'pokemonesFiltrados'])
  },
  methods: {
    ...mapActions(['fetchPokemones', 'fetchDetalle']),
    ...mapMutations(['ADD_FAVORITO', 'REMOVE_FAVORITO', 'UPDATE_FAVORITO'])
  }
}

Manejo de errores en Actions

fetchPokemones({ commit }) {
  commit('SET_LOADING', true)
  commit('SET_ERROR', null)
  axios.get('https://pokeapi.co/api/v2/pokemon?limit=20')
    .then(res => {
      // ... procesar datos
    })
    .catch(err => {
      if (err.response) {
        commit('SET_ERROR', `Error ${err.response.status}`)
      } else if (err.request) {
        commit('SET_ERROR', 'Sin conexión al servidor')
      } else {
        commit('SET_ERROR', err.message)
      }
    })
    .finally(() => commit('SET_LOADING', false))
}

Plugin de persistencia — Favoritos

// store/plugins/persistence.js
const persistencePlugin = (store) => {
  // Al iniciar: restaurar favoritos guardados
  const saved = localStorage.getItem('pokemon-favoritos')
  if (saved) {
    try {
      JSON.parse(saved).forEach(fav => {
        store.commit('ADD_FAVORITO', fav)
      })
    } catch (e) {
      localStorage.removeItem('pokemon-favoritos')
    }
  }

  // En cada mutation de favoritos: guardar
  store.subscribe((mutation, state) => {
    if (mutation.type.includes('FAVORITO')) {
      localStorage.setItem(
        'pokemon-favoritos',
        JSON.stringify(state.favoritos)
      )
    }
  })
}

export default persistencePlugin

Pinia — La evolución de Vuex

Vuex

State, Getters, Mutations, Actions. Mutations obligatorias. commit + dispatch.

Pinia

State, Getters, Actions. Sin mutations: se muta directo con this.count++.

Vue 2 → Vuex 3. Si migran a Vue 3, Pinia es la recomendación oficial. Los conceptos son los mismos.

Errores comunes

Axios en mutations

Las mutations son síncronas. Axios va siempre en actions.

No manejar loading

Sin SET_LOADING la UI no muestra feedback. Siempre: true → operación → false.

Mutar state directo

Nunca this.favoritos.push() desde el componente. Siempre commit.

Persistir datos sensibles

No guardes tokens ni contraseñas en localStorage con el plugin.
